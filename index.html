function binarySearch(A, x){
  let start = 0;
  let end = A.length-1;
  while(start<=end) {
    let middle = Math.floor( (start+end)/2 );
    if (A[middle]===x) {
      return middle;
    }
    else {
      if(A[middle] > x){
        end = middle-1;
      }
      else{
        start = middle+1;
      }
    }
  }
  return -1;
}
